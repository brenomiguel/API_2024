CREATE DATABASE biblioteca;

USE biblioteca;

CREATE TABLE usuario (
    id_usuario int PRIMARY KEY,
    nome varchar(100),
    email varchar(100),
    senhas varchar(255 ),
    endereco varchar(255),
    telefone varchar(20),
    tipo_usuario enum('admin', 'leitor')
);

CREATE TABLE livros (
    id_livro int PRIMARY KEY,
    titulo varchar(255),
    autor varchar(100),
    quantidade_fisico int,
    url_ebook varchar(255),
    tipo_livro enum('fisico', 'ebook'),
    editora varchar(100),
    ano_publicacao int(4),
    fk_emprestimos_id_emprestimo int
);

CREATE TABLE emprestimos (
    id_emprestimo int PRIMARY KEY,
    id_usuario int,
    id_livro int,
    data_devolucao date,
    data_emprestimo date,
    status enum('emprestado', 'devolvido'),
    fk_usuario_id_usuario int
);

CREATE TABLE reservas (
    id_reserva int PRIMARY KEY,
    id_usuario int,
    id_livro int,
    data_reserva date,
    status enum('ativa', 'cancelada', 'atendida'),
    fk_usuario_id_usuario int
);

CREATE TABLE ter (
    fk_reservas_id_reserva int,
    fk_livros_id_livro int
);
